# Local-Mesh Distributed Orchestration Implementation Plan

## Overview
Transform the existing Redis-based queue system into a P2P-capable local-mesh orchestration system that enables "home-lab" scaling. This plan extends the current architecture with:
- Automatic worker discovery via mDNS/Zeroconf
- Resource-aware task scheduling based on worker capabilities
- Cross-node checkpoint replication for zero data loss

---

## Current State Analysis

### Existing Infrastructure
1. **Queue System** (`src/infrastructure/queue/`):
   - Redis-backed job queue with priority scheduling
   - `JobQueue` class with Lua script atomic operations
   - `Worker` class with heartbeat, job processing, graceful shutdown
   - `WorkerInfo` model with basic metadata (hostname, pid, status, concurrency)

2. **Checkpointing** (`src/core/checkpoint.py`):
   - `CheckpointManager` with local filesystem storage
   - `CheckpointState` with stage results, metrics, iterative state
   - No cross-node replication capability

3. **Orchestrator** (`src/pipeline/services/pipeline_orchestrator/`):
   - `PipelineOrchestrator` with stage-based execution
   - Uses `TaskEnvelope` for job payloads
   - No resource-aware scheduling

### Gaps to Address
1. No P2P worker discovery (requires manual Redis config)
2. No resource profiling (CPU/RAM detection)
3. No task-to-worker matching based on capabilities
4. Checkpointing is local-only (no failover)

from pydantic import BaseModel, Field
import psutil

class ResourceProfile(BaseModel):
    """System resource information for a worker."""
    cpu_count: int = Field(default_factory=lambda: psutil.cpu_count(logical=True) or 1)
    cpu_freq_mhz: float = Field(default=0.0)
    total_ram_mb: int = Field(default_factory=lambda: psutil.virtual_memory().total // 1024 // 1024)
    available_ram_mb: int = Field(default=0)
    disk_gb_free: float = Field(default=0.0)
    platform: str = Field(default="unknown")
    python_version: str = Field(default="unknown")
    
    @classmethod
    def detect(cls) -> ResourceProfile:
        """Detect current system resources."""
        import platform
        mem = psutil.virtual_memory()
        cpu_freq = psutil.cpu_freq()
        disk = psutil.disk_usage('/')
        
        return cls(
            cpu_count=psutil.cpu_count(logical=True) or 1,
            cpu_freq_mhz=cpu_freq.max if cpu_freq else 0.0,
            total_ram_mb=mem.total // 1024 // 1024,
            available_ram_mb=mem.available // 1024 // 1024,
            disk_gb_free=disk.free / (1024**3),
            platform=platform.system(),
            python_version=platform.python_version(),
        )

class WorkerInfo(BaseModel):

    # ... existing fields ...

    capabilities: list[str] = Field(default_factory=list)  # e.g., ["browser", "heavy_compute"]

class TaskResourceRequirement(BaseModel):
    """Resource requirements for a task type."""
    min_cpu_cores: int = Field(default=1, ge=1)
    min_ram_mb: int = Field(default=256, ge=128)
    requires_browser: bool = Field(default=False)
    requires_gpu: bool = Field(default=False)

    @classmethod
    def for_task_type(cls, task_type: str) -> TaskResourceRequirement:
        """Return resource requirements based on task type."""
        # Define task categories
        heavy_tasks = {"headless_browser", "dom_xss", "websocket_hijacking"}
        light_tasks = {"port_probe", "http_methods", "cloud_metadata"}
        
        if task_type in heavy_tasks:
            return cls(min_cpu_cores=2, min_ram_mb=2048, requires_browser=True)
        elif task_type in light_tasks:
            return cls(min_cpu_cores=1, min_ram_mb=256, requires_browser=False)
        return cls()  # default medium

### Phase 3: Resource-Aware Scheduler

**Goal**: Implement a scheduler that matches tasks to workers based on resource availability.

#### 3.1 Create Resource-Aware Scheduler
**File**: `src/infrastructure/scheduling/resource_aware.py`

```python
"""Resource-aware scheduler for matching tasks to workers."""

import logging
from typing import Any

from src.infrastructure.queue.models import Job, TaskResourceRequirement, WorkerInfo

logger = logging.getLogger(__name__)

class ResourceAwareScheduler:
    """Schedules jobs based on worker resource availability."""
    
    def __init__(self) -> None:
        self.workers: dict[str, WorkerInfo] = {}
    
    def update_worker(self, worker_id: str, info: WorkerInfo) -> None:
        """Update worker information."""
        self.workers[worker_id] = info
    
    def select_worker(self, job: Job) -> str | None:
        """Select the best worker for a job based on resource requirements."""
        requirements = TaskResourceRequirement.for_task_type(job.type)
        
        candidates = []
        for worker_id, worker in self.workers.items():
            if self._can_handle(worker, requirements):
                score = self._calculate_score(worker, requirements)
                candidates.append((score, worker_id))
        
        if not candidates:
            return None
        
        # Sort by score (higher is better)
        candidates.sort(reverse=True)
        return candidates[0][1]
    
    def _can_handle(self, worker: WorkerInfo, req: TaskResourceRequirement) -> bool:
        """Check if a worker can handle the given requirements."""
        if worker.resources.cpu_count < req.min_cpu_cores:
            return False
        if worker.resources.available_ram_mb < req.min_ram_mb:
            return False
        if req.requires_browser and "browser" not in worker.capabilities:
            return False
        if worker.status not in ("idle", "busy"):
            return False
        return True
    
    def _calculate_score(self, worker: WorkerInfo, req: TaskResourceRequirement) -> float:
        """Calculate a suitability score for a worker."""
        score = 0.0
        # More available RAM = better
        score += worker.resources.available_ram_mb / 1024.0 * 10
        # More CPU cores = better
        score += worker.resources.cpu_count * 5
        # Fewer active jobs = better
        score -= len(worker.active_jobs) * 20
        return score
```

### Phase 4: Cross-Node Checkpoint Replication

**Goal**: Replicate checkpoint state across nodes so any worker can resume failed tasks.

#### 4.1 Create Distributed Checkpoint Store
**File**: `src/infrastructure/checkpoint/distributed.py`

```python
"""Distributed checkpoint storage using Redis for cross-node replication."""

import json
import logging
import time
from typing import Any

from src.core.checkpoint import CheckpointState
from src.infrastructure.queue.redis_client import RedisClient

logger = logging.getLogger(__name__)

class DistributedCheckpointStore:
    """Stores checkpoint state in Redis for access by any worker."""
    
    CHECKPOINT_KEY_PREFIX = "checkpoint:run:"
    LEASE_KEY_PREFIX = "checkpoint:lease:"
    
    def __init__(self, redis_client: RedisClient, node_id: str):
        self.redis = redis_client
        self.node_id = node_id
    
    async def save_checkpoint(self, state: CheckpointState, worker_id: str) -> bool:
        """Save checkpoint state to Redis with lease ownership."""
        key = f"{self.CHECKPOINT_KEY_PREFIX}{state.pipeline_run_id}"
        data = json.dumps(state.to_dict())
        
        # Use Redis pipeline for atomic operation
        pipe = self.redis.pipeline()
        pipe.hset(key, "state", data)
        pipe.hset(key, "owner", worker_id)
        pipe.hset(key, "updated_at", str(time.time()))
        pipe.expire(key, 86400)  # 24 hour TTL
        pipe.execute()
        
        logger.info("Saved checkpoint for run %s (owner=%s)", state.pipeline_run_id, worker_id)
        return True
    
    async def load_checkpoint(self, run_id: str) -> CheckpointState | None:
        """Load checkpoint state from Redis."""
        key = f"{self.CHECKPOINT_KEY_PREFIX}{run_id}"
        data = self.redis.execute_command("HGET", key, "state")
        
        if not data:
            return None
        
        state_dict = json.loads(data)
        return CheckpointState.from_dict(state_dict)
    
    async def take_ownership(self, run_id: str, new_worker_id: str, timeout: float = 300.0) -> bool:
        """Take ownership of a checkpoint (for failover)."""
        key = f"{self.CHECKPOINT_KEY_PREFIX}{run_id}"
        lease_key = f"{self.LEASE_KEY_PREFIX}{run_id}"
        
        # Atomic compare-and-swap using Lua script
        script = """
        local current_owner = redis.call('HGET', KEYS[1], 'owner')
        if current_owner == ARGV[1] or current_owner == false then
            redis.call('HSET', KEYS[1], 'owner', ARGV[2])
            redis.call('SET', KEYS[2], ARGV[2], 'EX', ARGV[3])
            return 1
        end
        return 0
        """
        result = self.redis.execute_script(script, keys=[key, lease_key], args=[new_worker_id, timeout])
        return result == 1
```

class Worker:
    async def _handle_stale_checkpoints(self) -> None:
        """Check for checkpoints owned by dead workers and take ownership."""
        dead_workers = self._discover_dead_workers()
        
        for worker_id in dead_workers:
            checkpoints = self._get_worker_checkpoints(worker_id)
            for run_id in checkpoints:
                # Take ownership and resume
                success = await self.distributed_store.take_ownership(run_id, self.worker_id)
                if success:
                    logger.info("Took ownership of checkpoint %s from dead worker %s", run_id, worker_id)
                    await self._resume_pipeline(run_id)
